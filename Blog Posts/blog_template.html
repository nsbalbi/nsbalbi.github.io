void split(boolean axis, int nSplits) {
  // axis True = x-axis, False = y-axis
  // nSplits self explainatory (2 splits is in half)

  percents = new float[nSplits + 1];  
  // list of percents at which children will be generated
  
  // generate child heights
  childHeights = new float[nSplits];
  for (int i = 0; i < nSplits; i++) {
    childHeights[i] = scale/20*i * 1.0/(layer+1);
  }

  if (axis) { // x-axis

    float[] newXs = new float[nSplits + 1];
    // generate new coordinates
    for (int i = 0; i <= nSplits; i++) {
      // split evenly into nSplit pieces
      newXs[i] = x1 + (float)i/nSplits*(x2-x1);  
      percents[i] = (float)i/nSplits;
    }

    // add new subdivisions
    for (int i = 0; i < nSplits; i++) {
      Div newChild = new Div(newXs[i], y1, newXs[i+1], y2,
       layer + 1, this, true, i, height + childHeights[i]);
      children.add(newChild);
    }

  } else {

    float[] newYs = new float[nSplits + 1];
    // generate new coordinates
    for (int i = 0; i <= nSplits; i++) {
      newYs[i] = y1 + (float)i/nSplits*(y2-y1);
      percents[i] = (float)i/nSplits;
    }

    // add new subdivisions
    for (int i = 0; i < nSplits; i++) {
      Div newChild = new Div(x1, newYs[i], x2, newYs[i+1], 
       layer + 1, this, false, i, height + childHeights[i]);
      children.add(newChild);
    }

  }

  ogPercents = percents.clone();  
  // clone (not reference) percents to record
  ogChildHeights = childHeights.clone();

}
